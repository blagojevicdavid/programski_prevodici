#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <HTTPClient.h>

USBHIDKeyboard Keyboard;

const char* ssid = "Yettel_EAC69C";
const char* pass = "4ThtCzfS";

//RAW LINK
const char* url = "https://raw.githubusercontent.com/blagojevicdavid/programski_prevodici/refs/heads/main/v10/config.txt";

String downloadFile(const char* link) {
  WiFiClientSecure client;
  client.setInsecure();

  HTTPClient https;
  if (!https.begin(client, link)) return "";

  int code = https.GET();
  if (code != 200) {
    https.end();
    return "";
  }

  String data = https.getString();
  https.end();
  return data;
}

void setup() {
  USB.begin();
  Keyboard.begin();

  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, pass);
  while (WiFi.status() != WL_CONNECTED) delay(200);

  String text = downloadFile(url);

  for (size_t i = 0; text[i] != '\0'; ++i) {
    Keyboard.write(text[i]);
    // opcioni delay ako računar ne stigne da obradi brze unose:
    delay(5);
  }
}
