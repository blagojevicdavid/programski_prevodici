3.(1) U tabeli Grad obrisati kolonu ZIPKOD koja predstavlja poštanski broj grada.

select p.idpr, p.nazpr, r.scena,
case
    when r.cenakrt <= 1500 then 'Niska'
    when r.cenakrt <= 2000 then 'Srednja'
    else 'visoka'
    end as kategorija
from predstava p join repertoar r on p.idpr = r.pre;
