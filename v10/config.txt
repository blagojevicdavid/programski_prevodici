7.(1.5) U tabeli Grad promeniti kolonu OZNAKA tako da sadrži prva tri slova naziva grada i „…“ (npr. za Beograd „Beo…“).

select po.idpz, po.nazpz, g.nazg, count(r.pre) as broj_predstava
from pozoriste po left join repertoar r on po.idpz = r.poz join grad g on g.idg = po.gradp 
group by po.idpz, po.nazpz, g.nazg
having count(r.pre) < 3;
8.(1.5) Prikazati šifru, naziv pozorišta, naziv grada u kojem se nalazi pozorište i broj predstava tog pozorišta.
Ako pozorište nema predstavu na repertoaru za broj predstava postaviti 0 (nula). Prikazati samo pozorišta koja imaju manje od 3 predstave na repertoaru.

create view Pogled_pozoriste_Cene as
select po.idpz, po.nazpz, nvl(min(r.cenakrt), 0) as min_cena, nvl(max(r.cenakrt), 0) as max_cena
from pozoriste po left join repertoar r on po.idpz = r.poz
group by po.idpz, po.nazpz;
9.(2) Kreirati pogled Pogled_Pozoriste_Cene koji ?e za svako pozorište (IDPZ, NAZPZ) prikazati minimalnu i
maksimalnu cenu karte. Ako pozorište nema predstavu na repertoaru, za minimalnu i maksimalnu cenu karte postaviti 0 (nula).

SELECT p.idpr, p.nazpr, r.cenakrt, r.brmes,
       ROUND(r.cenakrt - (SELECT AVG(r2.cenakrt) FROM repertoar r2 WHERE r2.poz = r.poz), 3 ) AS razlika
FROM predstava p
JOIN repertoar r ON p.idpr = r.pre
JOIN pozoriste po ON po.idpz = r.poz
WHERE r.cenakrt > (SELECT AVG(r2.cenakrt) FROM repertoar r2 WHERE r2.poz = r.poz)
ORDER BY r.brmes ASC;
