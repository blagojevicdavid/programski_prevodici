UPUTSTVO:
1. U Visual Studio kreiraj novi prazan ASP.NET Web Application (.NET Framework)
   projekat sa MVC opcijom. Naziv projekta: SOV2
   (File > New Project > ASP.NET Web Application (.NET Framework) > Empty + MVC)

2. U projektu kreiraj sledecu strukturu fajlova i kopiraj sadrzaj ispod.

3. NAPOMENA o namespace-u: Ako tvoj projekat ima drugaciji naziv,
   zameni svuda "SOV2" sa nazivom tvog projekta.

4. Pokreni projekat (F5) - otvara se Index stranica sa formom za unos.

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace SOV2.Models
{
    public class MobilniTelefon
    {
        public string Id { get; set; }
        public string Procesor { get; set; }
        public string VelicinaBaterije { get; set; }
        public Proizvodjac Proizvodjac { get; set; }
        public OperativniSistem OperativniSistem { get; set; }

        public MobilniTelefon() { }

        public MobilniTelefon(string id, string procesor, string velicinaBaterije,
                              Proizvodjac proizvodjac, OperativniSistem operativniSistem)
        {
            Id = id;
            Procesor = procesor;
            VelicinaBaterije = velicinaBaterije;
            Proizvodjac = proizvodjac;
            OperativniSistem = operativniSistem;
        }
    }
}

FAJL: Controllers/HomeController.cs
========================
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;
using SOV2.Models;

namespace SOV2.Controllers
{
    public class HomeController : Controller
    {
        // GET: Home - pocetna stranica sa formom za unos
        public ActionResult Index()
        {
            return View();
        }

        // POST: Home/Dodaj - dodavanje novog mobilnog telefona
        [HttpPost]
        public ActionResult Dodaj(MobilniTelefon telefon)
        {
            List<MobilniTelefon> telefoni = (List<MobilniTelefon>)HttpContext.Application["telefoni"];

            // Provera da li telefon sa tim Id-em vec postoji
            MobilniTelefon postojeci = telefoni.Find(t => t.Id.Equals(telefon.Id));
            if (postojeci != null)
            {
                ViewBag.Greska = "Mobilni telefon sa tim id-em vec postoji!";
                return View("Index");
            }

            telefoni.Add(telefon);
            HttpContext.Application["telefoni"] = telefoni;

            return RedirectToAction("Spisak");
        }

        // GET: Home/Spisak - tabelarni prikaz svih telefona
        public ActionResult Spisak()
        {
            List<MobilniTelefon> telefoni = (List<MobilniTelefon>)HttpContext.Application["telefoni"];
            return View(telefoni);
        }

        // GET: Home/Filtriraj?proizvodjac=Apple - filtriranje po proizvodjacu
        public ActionResult Filtriraj(string proizvodjac)
        {
            List<MobilniTelefon> telefoni = (List<MobilniTelefon>)HttpContext.Application["telefoni"];

            if (string.IsNullOrEmpty(proizvodjac))
            {
                return View("Spisak", telefoni);
            }

            Proizvodjac odabrani = (Proizvodjac)Enum.Parse(typeof(Proizvodjac), proizvodjac);
            List<MobilniTelefon> filtrirani = telefoni.FindAll(t => t.Proizvodjac == odabrani);

            return View("Spisak", filtrirani);
        }
    }
}

FAJL: Global.asax.cs   (PREPRAVITI postojeci Application_Start)
======================
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;
using System.Web.Optimization;
using System.Web.Routing;
using SOV2.Models;

namespace SOV2
{
    public class MvcApplication : System.Web.HttpApplication
    {
        protected void Application_Start()
        {
            AreaRegistration.RegisterAllAreas();
            FilterConfig.RegisterGlobalFilters(GlobalFilters.Filters);
            RouteConfig.RegisterRoutes(RouteTable.Routes);
            BundleConfig.RegisterBundles(BundleTable.Bundles);

            // Inicijalna lista mobilnih telefona (test podaci sa Slike 2)
            List<MobilniTelefon> telefoni = new List<MobilniTelefon>
            {
                new MobilniTelefon("1", "A15 Bionic", "2800 mAh",
                                   Proizvodjac.Apple, OperativniSistem.iOS),
                new MobilniTelefon("2", "Snapdragon 845", "3500 mAh",
                                   Proizvodjac.Samsung, OperativniSistem.Android),
                new MobilniTelefon("3", "Snapdragon 865", "4500 mAh",
                                   Proizvodjac.Xiaomi, OperativniSistem.Android)
            };

            HttpContext.Current.Application["telefoni"] = telefoni;
        }
    }
}

NAPOMENE:
================================================================================

1. Default ruta (App_Start/RouteConfig.cs) - NE diraj, ostavi default.
   Default je { controller = "Home", action = "Index" } sto je tacno ono
   sto nam treba (otvara Index.cshtml na pocetnoj).

2. Layout - mozes ostaviti default _Layout.cshtml ili obrisati liniju u
   _ViewStart.cshtml ako zelis ciste stranice. Test za zadatak je da
   funkcionalnost radi.

3. Test podaci (A15 Bionic / Snapdragon 845 / Snapdragon 865) ucitavaju se
   automatski pri pokretanju aplikacije iz Global.asax.cs - tako da kad
   kliknes "Dodaj" prvi put i odes na spisak vec ces videti 3 telefona.

4. Tok aplikacije:
   - Otvara se Index.cshtml (forma za unos + filter)
   - Klik na "Dodaj" -> POST na /Home/Dodaj
     * Ako Id postoji -> ostaje na Index sa porukom (Slika 3)
     * Ako Id ne postoji -> RedirectToAction na Spisak (Slika 2)
   - Klik na "Filtriraj" -> GET na /Home/Filtriraj?proizvodjac=X
     * Vraca View "Spisak" sa filtriranom listom (Slika 4)
   - Klik na "Nazad" na spisku -> vraca na Index.cshtml

5. Validacija - tekst poruke "Mobilni telefon sa tim id-em vec postoji!"
   je tacno onako kako je trazeno na Slici 3.

6. Sva polja se salju kao stringovi/enumi - automatic model binding ce
   da popuni MobilniTelefon objekat.
